MODULE create_and_transfer OUTPUT.

    " Container와 ALV 객체 생성

    " 레이아웃 설정
    PERFORM set_alv_layout_0100.

    PERFORM set_alv_event_0100.

  ELSE.
    CALL METHOD go_alv_grid->refresh_table_display( ).
