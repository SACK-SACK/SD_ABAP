MODULE status_0100 OUTPUT.
  SET PF-STATUS 'S0100'.

  CASE my_tabstrip-activetab.
    WHEN 'TAB1'.
      SET TITLEBAR 'T0100' WITH TEXT-t01.

    WHEN 'TAB2'.
      SET TITLEBAR 'T0100' WITH TEXT-t02.

    WHEN OTHERS.

      SET TITLEBAR 'T0100' WITH TEXT-t01.
  ENDCASE.

ENDMODULE.

MODULE fill_dynnr_0100 OUTPUT.
  CASE my_tabstrip-activetab.
    WHEN 'TAB1'.
      " 첫 번째 탭의 Subscreen 번호는 0110
      dynnr = '0110'.

    WHEN 'TAB2'.
      " 두 번째 탭의 Subsscreen 번호는 0120
      dynnr = '0120'.

    WHEN OTHERS.
      "초기 기본값 설정
      my_tabstrip-activetab = 'TAB1'.
      dynnr = '0110'.
  ENDCASE.

ENDMODULE.

MODULE init_alv_0120 OUTPUT.

MODULE set_input_field_0100 OUTPUT.

    IF screen-name EQ 'ZCA_SDT050-VBELN'.

    ENDIF.
